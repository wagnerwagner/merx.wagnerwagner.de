Text:

To make sure you get all the data you need from your customers, Merx uses Kirby’s page validation. You can define the required fields and set `validate` options for each field you want to validate. See [Kirby’s documentation](https://getkirby.com/docs/reference/system/validators) for all available validators. If you need more specific validation, you can [create your own validators](https://getkirby.com/docs/reference/plugins/extensions/validators).

## Define fields

Set up the fields, that are required to fulfill orders. Therefore create a new personal data section. If you don’t create you own personal data section, the [default personal data section](/reference/blueprints/sections/personal-data) is used.

<code caption="/site/blueprints/sections/personal-data.yml">
```yaml
type: fields
fields:
  email:
    label: Email
    type: email
    required: true
  given-name:
    label: Given name
    type: text
    required: true
    validate:
      minLength: 2
  family-name:
    label: Family name
    type: text
    required: true
  organization: # Optional field
    label: Organization
    type: text
# …
```
</code>

### Example response

When there are no validation errors, the API endpoint returns a redirect response, which redirects to the payment provider or directly to the order page.

When the validation fails, the API returns a JSON with one or more error messages per invalid field.

<code caption="POST https://shop.test/api/shop/checkout">
```json
{
	"status": "error",
	"message": "Field validation failed.",
	"code": 400,
	"key": "error.merx.fieldsvalidation",
	"details": {
		"email": {
			"label": "Email",
			"message": {
				"required": "Please enter something"

		}

}
```
</code>

## Use JavaScript to handle form submit

Use JavaScript to interfere the submit event of the checkout form. Use the validation details to show the error messages to the user.

<code caption="script.js">
```js
const checkoutForm = document.getElementById('checkout-form');
checkoutForm.addEventListener('submit', async (event) => {
  event.preventDefault();
  const response = await fetch(checkoutForm.action, {
    method: 'post',
    body: new FormData(checkoutForm),
  });
  console.log(response);

  if (response.redirected === true) {
    // [1] Redirect user to payment provider or directly to order page
    window.location.href = response.url;
  } else {
    // [2] Handle validation errors
    const json = await response.json();
    const errorDetails = json.details;
    console.log(errorDetails);
  }
});
```
</code>


### Complete validation handling


<code caption="script.js">
```js
const checkoutForm = document.getElementById('checkout-form');

/**
 * Creates a p element with class “error”.
 * @param {String} error e.g. “Please enter a valid email address”
 */
function createErrorElement(errorMessage) {
  return Object.assign(document.createElement('p'), {
    classList: 'error',
    innerText: errorMessage,
  });
}

/**
 * Removes all elements with class “error”.
 */
function clearErrors() {
  checkoutForm.querySelectorAll('.error').forEach((_) => _.remove());
}

checkoutForm.addEventListener('submit', async (event) => {
  event.preventDefault();

  clearErrors();

  const formData = new FormData(checkoutForm);

  // Disables all form fields to prevent multiple submissions
  [...checkoutForm.elements].forEach(_ => _.toggleAttribute('disabled', true));

  // Send form data tot API endpoint
  const response = await fetch(checkoutForm.action, {
    method: 'post',
    body: formData,
  });

  // Enables all form fields
  [...checkoutForm.elements].forEach(_ => _.toggleAttribute('disabled', false));

  console.log(response);

  if (response.redirected === true) {
    // Redirect user to payment provider or directly to order page
    window.location.href = response.url;
  } else {
    // Handle validation errors
    try {
      /** @see https://merx.wagnerwagner.de/reference/api/routes/shop-checkout-post */
      const json = await response.json();

      if (json.key === 'error.merx.fieldsvalidation') {
        // Handle field errors
        const errorDetails = json.details;

        for (const [key, error] of Object.entries(errorDetails)) {
          const labelElement = checkoutForm.querySelector(`label:has([name="${key}"])`);

          if (labelElement) {
            for (const errorMessage of Object.values(error.message)) {
              // Display error message(s) inside of label element
              labelElement.appendChild(createErrorElement(errorMessage));
            }
          } else {
            throw json.message ?? 'Fatal error';
          }
        }
      } else {
        throw json.message ?? 'Fatal error';
      }
    } catch (errorMessage) {
      // Display general error message before submit button
      checkoutForm.insertBefore(createErrorElement(errorMessage), event.submitter)
    }
  }
});
```
</code>
